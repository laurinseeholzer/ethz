\section{Number Theory}

\subsection{Core Concepts}

\begin{definition}[Modular Arithmetic]
    Let $a, b \in \mathbb{Z}$ and $n \in \mathbb{Z}^+$. We say $a \equiv b \pmod n$ if $n \mid (a-b)$.
\end{definition}

\begin{theorem}[Euler's Totient]
    If $a$ and $n$ are coprime, then:
    $$ a^{\varphi(n)} \equiv 1 \pmod n $$
\end{theorem}

\subsection{Key Proofs \& Lemmas}

\begin{lemma}[Euclid's Lemma]
    If a prime $p$ divides $ab$, then $p \mid a$ or $p \mid b$.
\end{lemma}

\section{Applications}

\subsection{Cryptography Basics}

\begin{example}
    Find $7^2 \pmod 5$. \\
    Since $7 \equiv 2 \pmod 5$, we have:
    $$ 7^2 \equiv 2^2 \equiv 4 \pmod 5 $$
\end{example}

\begin{codeexample}[language=Python]
def gcd(a, b):
    while b:
        a, b = b, a % b
    return a
\end{codeexample}

\begin{algorithm}[language=C]
int factorial(int n) {
    if (n == 0) return 1;
    return n * factorial(n - 1);
}
\end{algorithm}
