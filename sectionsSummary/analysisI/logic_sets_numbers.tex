\section{Logic, Sets and Numbers}

\subsection{Logic}

\begin{definition}
A \textbf{mathematical statement} is an assertion that is strictly true or false. Its negation is $\neg A \coloneqq A \text{ is false}$.
A statement depending on variables is a \textbf{predicate}.
\end{definition}

\begin{example}
$$P(x) \coloneqq x > 2$$
\end{example}

\begin{definition}
Quantifiers are short hand notation for making statements about predicates.
$$\text{forall}: \forall x \in \mathbb{R}: P(x)$$
$$\text{exists}: \exists x \in \mathbb{R}: P(x)$$
$$\text{exists unique}: \exists ! x \in \mathbb{R}: P(x)$$
$$\text{not exists}: \nexists x \in \mathbb{R}: P(x)$$
\end{definition}

\begin{important}
The oderder of quantifiers matters. $\forall x \exists y$ is not equivalent to $\exists y \forall x$.
\end{important}

\begin{definition}
Basic logical connectives are defined as:
$$A \land B \coloneqq A \text{ and } B \text{ are true}$$
$$A \lor B \coloneqq A \text{ or } B \text{ is true}$$
$$A \implies B \coloneqq A \text{ implies } B \text{ is true}$$
$$A \iff B \coloneqq A \implies B \land B \implies A$$
Observe that $\neg A \lor B \iff A \implies B$.
\end{definition}

\begin{theorem}
The following equivalences hold:
$$\neg (\forall x P(x)) \iff \exists x \neg P(x)$$
$$\neg (\exists x P(x)) \iff \forall x \neg P(x)$$
aswell as
$$\forall x (P(x) \land Q(x)) \iff (\forall x P(x)) \land (\forall x Q(x))$$
$$\exists x (P(x) \lor Q(x)) \iff (\exists x P(x)) \lor (\exists x Q(x))$$
\end{theorem}

\subsection{Sets}


\begin{definition}
A \textbf{set} is a collection of elements. The empty set $\emptyset$ contains no elements. $x \in A$ ($x \notin A$) means $x$ is (not) in $A$.
$A$ is a subset of $B$ denoted by $A \subseteq B$ iff. $$\forall x (x \in A \implies x \in B)$$
$A$ is a proper subset of $B$ denoted by $A \subset B$ iff. $$A \subseteq B \land A \neq B$$
\end{definition}

\begin{definition}
The \textbf{union} of $A$ and $B$ is defined as $A \cup B \coloneqq \{x \mid x \in A \lor x \in B\}$.
\end{definition}

\begin{definition}
The \textbf{intersection} of $A$ and $B$ is defined as $A \cap B \coloneqq \{x \mid x \in A \land x \in B\}$.
\end{definition}

\begin{definition}
The \textbf{difference} of $A$ and $B$ is defined as $A \setminus B \coloneqq \{x \mid x \in A \land x \notin B\}$.
\end{definition}

\begin{definition}
The \textbf{Cartesian product} of $A$ and $B$ is defined as $A \times B \coloneqq \{(a, b) \mid a \in A \land b \in B\}$.
\end{definition}

\subsection{Numbers}

The basic and often used sets of numbers are:

\begin{itemize}
    \item $\mathbb{N}$: The set of natural numbers 1, 2, 3, 4, ...
    \item $\mathbb{Z}$: The set of integers ... -2, -1, 0, 1, 2, ...
    \item $\mathbb{Q}$: The set of rational numbers $\frac{p}{q}$ where $p, q \in \mathbb{Z}$
    \item $\mathbb{R}$: The set of real numbers
\end{itemize}

\begin{remark}
Some extensions of the real numbers are 
$$\mathbb{R}^{+}_{0} / \mathbb{R}^{+} := [0, \infty) / (0, \infty)$$
$$\mathbb{R}^{-}_{0} / \mathbb{R}^{-} := (-\infty, 0] / (-\infty, 0)$$
\end{remark}

\begin{definition}
Continuous subsets are called \textbf{intervalls.} An intervall is called\\
\end{definition}
\begin{definition}
\textbf{open} if it does not contain its endpoints 
$$I = (a, b) \coloneqq \{x \mid a \lt x \lt b\}$$
\end{definition}
\begin{definition}
\textbf{closed} if it contains its endpoints 
$$I = [a, b] \coloneqq \{x \mid a \leq x \leq b\}$$
\end{definition}
\begin{definition}
\textbf{half open} if it contains one endpoint and not the other 
$$I = [a, b) \coloneqq \{x \mid a \leq x \lt b\}$$
\end{definition}
\begin{definition}
An intervall is \textbf{bounded} if it is bounded from above and below hence $a$ and $b$ are finite. A bounded and closed intervall is \textbf{compact}.
\end{definition}

\begin{definition}
An \textbf{upper (lower) bound} of a subset $X \subset \mathbb{R}$ is an element $y \in \mathbb{R}$ such that:
$$\forall x \in X: x \leq (\geq) y$$
\end{definition}
\begin{definition}
The \textbf{maximum (minimum)} of a subset $X \subset \mathbb{R}$ is the element $x_{0} \in X$ such that:
$$\forall x \in X: x \leq (\geq) x_{0}$$
\end{definition}

\begin{definition}
Least upper bound and greatest lower bound are also called Supremum and Infimum.
\\ \\
The \textbf{Supremum} $sup(X)$ of a subset $X \subset \mathbb{R}$ is the least upper bound of $X$. and can be characterized by:
$$\forall x \in X: x \leq sup(X) \land$$
$$ \forall \epsilon \gt 0: \exists x \in X: x \gt sup(X) - \epsilon$$

The \textbf{Infimum} $inf(X)$ of a subset $X \subset \mathbb{R}$ is the greatest lower bound of $X$. and can be characterized by:
$$\forall x \in X: x \geq inf(X) \land$$
$$\forall \epsilon \gt 0: \exists x \in X: x \lt inf(X) + \epsilon$$
\end{definition}

\begin{theorem}
The \textbf{maximum and minimum are unique} aslong as they exist.
\\
\\
Every \textbf{non empty, bounded from above (below)} subset of $\mathbb{R}$ has a \textbf{maximum (minimum)}.
\end{theorem}

\begin{definition}
A set $X$ is said to be (order)complete if for every non empty subsets $A, B \subseteq X$
$$\forall a \in A \forall b \in B: a \leq b$$
then there there exists an element $c \in X$ such that:
$$\forall a \in A: a \leq c \land \forall b \in B: c \leq b$$
\end{definition}

\begin{theorem}
The \textbf{Archimedian Principle} says that
$$\forall x \in \mathbb{R}, \forall y \gt 0 \exists n \in \mathbb{N}: n y \gt x$$
$$\forall \epsilon \gt 0 \exists n \in \mathbb{N}: \frac{1}{n} \lt \epsilon$$
\end{theorem}

\begin{theorem}
The Young Inequality states that for all $\epsilon \gt 0$ and all $x, y \in \mathbb{R}$:
$$2\lvert xy \rvert \leq \epsilon x^2 + \frac{1}{\epsilon} y^2$$
\end{theorem}

\begin{theorem}
$\mathbb{R}$ has order axioms: reflexivity, transitivity, antisymmetry, totality, making it an \textbf{ordered and order-complete field.}
\end{theorem}

\begin{definition}
For $x, y \in \mathbb{R}^n$, the standard scalar product is defined as:
$$x \cdot y = \langle x, y \rangle \coloneqq \sum_{i=1}^n x_i y_i$$
\end{definition}
\begin{definition}
The Euclidean norm and distance are defined as:
$$||x|| \coloneqq \sqrt{\sum_{i=1}^n x_i^2}$$
$$d(x, y) \coloneqq \sqrt{\sum_{i=1}^n (x_i - y_i)^2}$$
\end{definition}

\begin{theorem}
The triangle inequality holds for all $x, y, z \in \mathbb{R}^n$:
$$||x - z|| \leq ||x - y|| + ||y - z||$$
\end{theorem}

\begin{definition}
The \textbf{complex numbers} $\mathbb{C}$ are numbers of the form:
$$z = a + ib \quad \text{where } a, b \in \mathbb{R}$$
$a$ is the real part $\text{Re}(z)$, and $b$ is the imaginary part $\text{Im}(z)$. The imaginary unit $i$ has the property $i^2 = -1$.
\end{definition}

\begin{definition}
Arithmetic of complex numbers $z_1 = x_1 + i y_1$ and $z_2 = x_2 + i y_2$: 
$$z_1 + z_2 = (x_1 + x_2) + i(y_1 + y_2)$$
$$z_1 \cdot z_2 = (x_1 x_2 - y_1 y_2) + i(x_1 y_2 + y_1 x_2)$$
\end{definition}
\begin{definition}
The complex conjugate of $z = x + iy$ is $\bar{z} = x - iy$. Division is calculated by expanding with the conjugate of the denominator: 
$$\frac{z}{w} = \frac{z \cdot \bar{w}}{w \cdot \bar{w}} = \frac{z \cdot \bar{w}}{|w|^2}$$
\end{definition}

\begin{remark}
Some usfull properties of the conjugate:
\begin{itemize}
    \item $z \cdot \bar{z} = |z|^2$
    \item $z + \bar{z} = 2\text{Re}(z)$ and $z - \bar{z} = 2i\text{Im}(z)$
    \item $\overline{z+w} = \bar{z} + \bar{w}$ and $\overline{z \cdot w} = \bar{z} \cdot \bar{w}$
\end{itemize}
\end{remark}

\begin{definition}
Geometrically, complex numbers can be represented in the Gaussian plane.
The \textbf{absolute value} of $z = a + ib$ is its \textbf{distance from the origin}:
$$|z| = \sqrt{a^2 + b^2}$$
The distance between $z_1$ and $z_2$ is $d = |z_1 - z_2|$.
The triangle inequality holds in $\mathbb{C}$: $|z + w| \leq |z| + |w|$.
\end{definition}

\begin{theorem}
The Euler's formula states that for $t \in \mathbb{R}$:
$$e^{it} = \cos(t) + i\sin(t)$$
This allows writing trigonometric functions using complex exponentials:
$$\cos(t) = \frac{e^{it} + e^{-it}}{2}, \quad \sin(t) = \frac{e^{it} - e^{-it}}{2i}$$
\end{theorem}

Eulers formula leads us to a new representation of complex numbers: the polar-coordinates.

\begin{definition}
For $z \in \mathbb{C}$, the polar-coordinates are defined as:
$$z = r(\cos(t) + i\sin(t)) = r e^{it}$$
where $r = |z|$ is the absolute value and $t = \arg(z)$ is the argument.
\end{definition}

Multiplication and division in polar-coordinates is much easier than in cartesian coordinates:
$$z_1 \cdot z_2 = r_1 e^{it_1} \cdot r_2 e^{it_2} = r_1 r_2 e^{i(t_1 + t_2)}$$
$$\frac{z_1}{z_2} = \frac{r_1 e^{it_1}}{r_2 e^{it_2}} = \frac{r_1}{r_2} e^{i(t_1 - t_2)} $$

So is exponentiation or taking roots:
$$z^n = (r e^{it})^n = r^n e^{int}$$
$$\sqrt[n]{z} = \sqrt[n]{r} e^{i\frac{t+2\pi k}{n}}, \quad k \in \{0, 1, \dots, n-1\}$$

\begin{definition}
The $n$-th roots of unity are the solutions to $z^n = 1$, which are given by:
$$z_k = e^{i\frac{2\pi k}{n}}, \quad k \in \{0, 1, \dots, n-1\}$$
\end{definition}

\begin{theorem}
The sum of the roots of unity is always zero.
$$\sum_{k=0}^{n-1} e^{i\frac{2\pi k}{n}} = 0$$
\end{theorem}

\begin{theorem}
The fundamental theorem of algebra states that every polynomial of degree $n \geq 1$ has exactly $n$ roots in $\mathbb{C}$ (counting multiplicity). These roots appear in konplex-conjugate pairs.
\end{theorem}
