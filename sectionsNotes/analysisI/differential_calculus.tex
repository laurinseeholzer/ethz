\section{Differential calculus}

\begin{definition}
    The \textbf{average rate of change} of a function $f$ in the domain between $a$ and $a + h$ is defined as
    $$ \frac{\Delta y}{\Delta x} = \frac{f(a+h)-f(a)}{h}$$
    and also called the differential quotient.
\end{definition}

\begin{definition}
    The \textbf{derivative} $f'$ of a function $f$ at the coordinate $a$ is defined as
    $$\lim_{h \to 0}\frac{f(a+h)-f(a)}{h} = lim_{b \to a}\frac{f(b)-f(a)}{b-a} = \frac{df}{dx}\vert_{x=a} = f'(a)$$
\end{definition}

\begin{remark}
    The derivative $f'(a)$ can geometrically interpreted as the incline/decline of the graph of $f$ at the point $(a, f(a))$.
\end{remark}

\begin{definition}
    If for a function $f$ at the cooridnate $a$ the derivative $\exists f'(a)$ then f is called \textbf{differentiable} at $a$. If $f$ is differentiable for every $a \in \mathbb{D}$ then $f$ is called differentiable.
\end{definition}

\begin{example}
    let $f(x) = |x|$ we can see that for $a=0$ there is no derivative (we have a sharp change in "direction")
\end{example}

\begin{remark}
    We can generally expect that there are no derivatives at undefined points $a \notin \mathbb{D}$.
\end{remark}

\begin{lemma}
    If $f$ is differentiable at $x_0$ then $f$ is continous at $x_0$.
\end{lemma}

\begin{proof}
    Assume $f$ differentiable then
    $$\lim_{x \to x_0}f(x) = \lim_{h \to 0} f(x_0+h) = \lim_{h \to 0} f(x_0 + h) - f(x_0) + f(x_0)$$
    $$ = f(x_0) + \lim_{h \to 0} f(x_0+h)-f(x_0) = f(x_0) + \lim_{h \to 0} \frac{f(x_0+h)-f(x_0)}{h} \cdot h$$
    $$ = f(x_0) + \lim_{h \to 0} h \cdot \lim_{h \to 0} \frac{f(x_0+h)-f(x_0)}{h}$$
    Observe that the last term exists by the assumption hence
    $$ = f(x_0) + 0 \cdot f'(x_0) = f(x_0)$$
\end{proof}

\begin{definition}
    If the limit 
    $$\lim_{h>0 \to 0}\frac{f(a+h)-f(a)}{h}$$
    existst then we call it the \textbf{right-sided derivative} (or the left-sided derivative analogue)
\end{definition}

\begin{definition}
    For a function $f$ we can iteratively define higher derivatives for $n \in \mathbb{N}_0$
    $$ f^(0) = f, f^(1) = f', f^(2) = f'', \dots f^(n+1) = (f^(n))'$$
    if $f^(n)$ exists we call it $n$-times differentiable.
\end{definition}

\begin{definition}
    if $f^(n)$ is continous we call $f$ $n$-times continous differentiable. The set of all $n$-times continous differentiable functions on $D$ is denoted as $C^n(D)$
\end{definition}

\begin{definition}
    We call $C^{\infty}(D) \coloneqq \cap_{n=0}^{\infty} C^n(D)$ the set of smooth functions.
\end{definition}

\subsection{Derivation rules}

\begin{theorem}
    For functions $f(x)$ we have
    \begin{itemize}
        \item $f(x) = a x^p \quad \implies \quad f'(x) = a p x^{p-1}$
        \item $f(x) = a^x \quad \implies \quad f'(x) = ln(a) \cdot a^x$
        \item $f(x) = sin(x) \quad \implies \quad f'(x) = cos(x)$
        \item $f(x) = cos(x) \quad \implies \quad f'(x) = -sin(x)$
        \item $f(x) = tan(x) \quad \implies \quad f'(x) = \frac{1}{cos^2(x)} = 1 + tan^2(x)$
        \item $f(x) = sinh(x) \quad \implies \quad f'(x) = cosh(x)$
        \item $f(x) = cosh(x) \quad \implies \quad f'(x) = sinh(x)$
        \item $f(x) = ln(x) \quad \implies \quad f'(x) = \frac{1}{x}$
        \item $f(x) = arcsin(x) \quad \implies \quad f'(x) = \frac{1}{\sqrt{1-x^2}}$
        \item $f(x) = arccos(x) \quad \implies \quad f'(x) = \frac{-1}{\sqrt{1-x^2}}$
        \item $f(x) = arctan(x) \quad \implies \quad f'(x) = \frac{1}{1+x^2}$
    \end{itemize}
\end{theorem}

\begin{theorem}
    \textbf{Sum Rule:} For the $n$-th derivative of a sum:
    $$(f + g)^{(n)} = f^{(n)} + g^{(n)} $$
\end{theorem}

\begin{proof}
    This can be proved inductively by applying the definition of the derivative $n$ times.
    $$(f+g)' (x) = lim_{h \to 0} \frac{f(x+h)+g(x+h)-f(x)-g(x)}{h}$$
    $$ = lim_{h \to 0} \frac{f(x+h)-f(x)}{h} + lim_{h \to 0} \frac{g(x+h)-g(x)}{h}$$
    $$ = f'(x) + g'(x)$$
    For $n=2$ we have
    $$(f+g)'' (x) = (f'+g')' (x) = f''(x) + g''(x)$$
    and so on.
\end{proof}

\begin{theorem}
    \textbf{General Leibniz Rule (Product):} The $n$-th derivative of a product is given by:
    $$(f \cdot g)^{(n)} = \sum_{k=0}^{n} \binom{n}{k} f^{(n-k)} g^{(k)} $$
\end{theorem}

\begin{proof}
    This is proved analogously to the sum rule.
    $$(f \cdot g)'(x) = lim_{h \to 0} \frac{f(x+h)g(x+h)-f(x)g(x)}{h}$$
    $$ = lim_{h \to 0} \frac{f(x+h)g(x+h)-f(x)g(x+h)+f(x)g(x+h)-f(x)g(x)}{h}$$
    $$ = lim_{h \to 0} \frac{g(x+h)(f(x+h)-f(x))}{h} + lim_{h \to 0} \frac{f(x)(g(x+h)-g(x))}{h}$$
    $$ = g(x)f'(x) + f(x)g'(x)$$
    For $n=2$ we have
    $$(f \cdot g)''(x) = (f'g+fg')'(x) = f''g+f'g'+f'g'+fg'' = f''g+2f'g'+fg''$$
    and so on.
\end{proof}

\begin{theorem}
    \textbf{Chain Rule:} For the composition of functions:
    $$(f \circ g)'(x) = f'(g(x)) \cdot g'(x) $$
\end{theorem}

\begin{proof}
    Observe that 
    $$g(y) = g(y_0) + g'(y_0)(y-y_0) + g(y)-g(y_0)-g'(y_0)(y-y_0)$$
    with the first two summands being a linear (tangential) approximation of $g(y)$ near $y_0$
    $$ g(y) \approx g(y_0) + g'(y_0)(y-y_0)$$ 
    By taking out $(y - y_0)$ of the last two summands in the original equation we get:
    $$g(y) = g(y_0) + g'(y_0)(y-y_0) + \left[\frac{g(y)-g(y_0)}{y-y_0}-g'(y_0)\right](y-y_0)$$
    We define $\epsilon = \frac{g(y)-g(y_0)}{y-y_0}-g'(y_0)$ and get
    $$\lim_{x \to x_0} \frac{g(f(x))-g(f(x_0))}{x-x_0}$$
    $$ = \lim_{x \to x_0} \left(f(f(x_0)) + g'(f(x_0))[f(x)-f(x_0)]+\epsilon f(x)[f(x)-f(x_0)] - g(f(x_0))\right) \frac{1}{x-x_0}$$
    $$ = g'(y_0)f'(x_0)$$
\end{proof}

\begin{theorem}
    \textbf{Quotient Rule:} For $g(x) \neq 0$:
    $$\left(\frac{f}{g}\right)' = \frac{f'g - fg'}{g^2} $$
\end{theorem}

\begin{remark}
    This follows directly from the product rule.
\end{remark}

\begin{theorem}
    \textbf{Inverse Functions:} If $f$ is differentiable and injective:
    $$(f^{-1})'(x) = \frac{1}{f'(f^{-1}(x))} $$
\end{theorem}

\subsection{Bernoulli-L'Hopital}

\begin{theorem}
    Let $I \subset \mathbb{R}$ be a an intervall, $f: I \to \mathbb{R}$, and let $x_0$ be a local Extrema of $f$. Then one of the following holds:
    \begin{itemize}
        \item $x_0$ is an endpoint of $I$
        \item $f$ is not differentiable at $x_0$
        \item $f$ is differentiable at $x_0$ and $f'(x_0) = 0$
    \end{itemize}
\end{theorem}

\begin{remark}
    From $f'(x_0) = 0$ we can not conclude that $x_0$ is an extremum.
\end{remark}

\begin{theorem}
    \textbf{Rolle's Theorem}: Let $a \lt b$ with $f(a) = f(b)$ and $f:[a, b] \to \mathbb{R}$ a completely continous function on $[a, b]$ and differentiable on $(a, b)$ then $\exists \xi \in (a, b)$ with $f'(\xi) = 0$
\end{theorem}

\begin{proof}
    As $[a,b]$ is compact we can conclud that there exists $\xi$ such that $f(\xi)$ is max or min hence $f'(\xi)=0$. If $\xi \in (a,b)$ we found our $\xi$ else the max and min is at $a$ therfore also at $b$ hence the function is constant which implies that $f'(x) = 0 \forall x \in [a,b]$.
\end{proof}

\begin{theorem}
    \textbf{Mean-Value-Theorem}: Let $a \lt b, f:[a, b] \to \mathbb{R}$ a completely continous function on $[a, b]$ and differentiable on $(a, b)$ then $\exists \xi \in (a, b)$ with
    $$f'(\xi) = \frac{f(b)-f(a)}{b-a}$$
\end{theorem}

\begin{proof}
    We define a function $g$ with 
    $$g(x) \coloneqq f(x) - \frac{f(b)-f(a)}{b-a} (x-a)$$
    and observe that $g$ fulfilles the prequisites to apply rolles theorem. Hence 
    $$\exists \xi g'(\xi) = 0$$
    $$\implies \exists \xi f'(\xi) - \frac{f(b)-f(a)}{b-a}\xi  = 0$$
    $$\implies \exists \xi f'(\xi) = \frac{f(b)-f(a)}{b-a}$$
\end{proof}

\begin{theorem}
    \textbf{Bernoulli-L'Hopital's Rule:} Let $f, g: I \to \mathbb{R}$ be differentiable functions on $(a, b)$ and let $x_0 \in (a, b)$. If $g(x) \neq 0$ and $g'(x) \neq 0$ for all $x \in (a, b)$. If 
    $$\lim_{x \to x_0} |f(x)| = \lim_{x \to x_0} |g(x)| = 0 \text{or} \infty \quad  \text{and} \quad \exists L = \lim_{x \to x_0} \frac{f'(x)}{g'(x)}$$
    $$\lim_{x \to x_0} \frac{f(x)}{g(x)} = \lim_{x \to x_0} \frac{f'(x)}{g'(x)} = L$$
\end{theorem}

\begin{example}
    Let $f(x) = sin(x)$ and $g(x) = x$ we have $sin(0) = 0$ and $g(0) = 0$ further we have $f'(x) = cos(x)$ and $g'(x) = 1 \neq 0$ hence 
    $$\lim_{x \to 0} \frac{sin(x)}{x} = \lim_{x \to 0} \frac{cos(x)}{1} = 1$$
\end{example}

\begin{remark}
    This example can also be solved via the definition of the derivative.
    $$\lim_{x \to 0} \frac{sin(x)}{x} = \lim_{x \to 0} \frac{sin(x)-sin(0)}{x-0} = sin'(0) = cos(0) = 1$$
\end{remark}

\subsection{Convex and Concave graphs}

\begin{theorem}
    Let $(a, b) \subset \mathbb{R}$ and $f: (a, b) \to \mathbb{R}$ be differentiable. Then 
    $$\forall x: f'(x) \geq 0 \quad \iff \quad f \text{ is increasing}$$
\end{theorem}

\begin{remark}
    If $f'(x) = 0$ for all $x$ then $f$ is constant.
\end{remark}

\begin{theorem}
    Let $(a, b) \subset \mathbb{R}$ and $f: (a, b) \to \mathbb{R}$ be differentiable. Then 
    $$f \text{ is convex} \quad \iff \quad f' \text{ is increasing}$$
\end{theorem}

\begin{theorem}
    Let $(a, b) \subset \mathbb{R}$ and $f: (a, b) \to \mathbb{R}$ be twice differentiable. Then 
    $$f \text{ is convex} \quad \iff \quad f'' \geq 0$$
\end{theorem}
