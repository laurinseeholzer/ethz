% --- Define Theorem Styles (tcolorbox for perfect breaking) ---

\newtheorem{definition}{Def.}
\newtheorem{theorem}[definition]{Thm.}
\newtheorem{lemma}[definition]{Lem.}
\newtheorem{remark}[definition]{Rem.}
\newtheorem{important}[definition]{Imp.}
\newtheorem{example}[definition]{Ex.}

% 1. Gray Block Style (Def, Thm, Lem)
\tcbset{
    grayBlock/.style={
        breakable,          % <-- Fixes the phantom blank pages
        enhanced,
        colback=black!5,    % Light gray background
        colframe=black!5,   % Matches background to hide border
        boxrule=0pt,
        arc=0pt,            % Sharp corners
        outer arc=0pt,
        top=3pt, bottom=3pt, left=4pt, right=4pt,
        boxsep=0pt,
        before skip=4pt, after skip=4pt,
        lines before break=1,
    },
    breakable,
    lines before break=1,
    before skip=0pt,        % Remove space before the box
    after skip=0pt,         % Remove space after the box
    boxsep=1pt,             % Shrink internal padding
    top=1pt, bottom=1pt,    % Shrink internal padding
}
\tcolorboxenvironment{definition}{grayBlock}
\tcolorboxenvironment{theorem}{grayBlock}
\tcolorboxenvironment{lemma}{grayBlock}

% 2. Standard Block Style (Remark - Left line only)
\tcbset{
    standardBlock/.style={
        breakable,
        enhanced,
        colback=white,
        colframe=black,
        boxrule=0pt,
        leftrule=1.5pt,     % Only the left line is drawn
        arc=0pt,
        outer arc=0pt,
        top=2pt, bottom=2pt, left=6pt, right=2pt,
        boxsep=0pt,
        before skip=4pt, after skip=4pt
    }
}
\tcolorboxenvironment{remark}{standardBlock}

% 3. Important Block Style (Full border)
\tcbset{
    importantBlock/.style={
        breakable,
        enhanced,
        colback=white,
        colframe=black,
        boxrule=1pt,        % Full 1pt border
        arc=0pt,
        outer arc=0pt,
        top=3pt, bottom=3pt, left=4pt, right=4pt,
        boxsep=0pt,
        before skip=4pt, after skip=4pt
    }
}
\tcolorboxenvironment{important}{importantBlock}


% --- Code & Algorithm Environments ---

\lstnewenvironment{codeexample}[1][]
{
    \refstepcounter{definition}
    \vspace{2pt} 
    \noindent\textbf{Code Example \thedefinition}\vspace{2pt} 
    \lstset{
        basicstyle=\ttfamily\scriptsize, 
        numbers=left,
        numberstyle=\tiny\color{black!50},
        stepnumber=1,
        numbersep=6pt,                  
        xleftmargin=18pt,               
        xrightmargin=2pt,               
        framexleftmargin=18pt,          
        framexrightmargin=2pt,          
        tabsize=4,
        showspaces=false,
        showstringspaces=false,
        breaklines=true,
        frame=single,
        rulecolor=\color{black!30},
        backgroundcolor=\color{black!2},
        aboveskip=2pt, belowskip=4pt,   
        breakatwhitespace=true,
        #1
    }
}
{}
