% --- Define Theorem Styles (tcolorbox for perfect breaking) ---

\theoremstyle{definition}           % Upright body text (not italic)
\newtheorem{definition}{Def.}
\newtheorem{theorem}[definition]{Thm.}
\newtheorem{lemma}[definition]{Lem.}
\newtheorem{remark}[definition]{Rem.}
\newtheorem{important}[definition]{Imp.}
\newtheorem{example}[definition]{Ex.}

% 1. Gray Block Style (Def, Thm, Lem, Ex)
% NOTE: Do NOT use 'breakable' — it conflicts with multicols*.
% Instead, keep boxes small (split large content across multiple environments).
\tcolorboxenvironment{definition}{
    enhanced jigsaw,        % jigsaw avoids the breakable overhead
    colback=black!5,
    colframe=black!5,
    boxrule=0pt,
    arc=0pt, outer arc=0pt,
    top=2pt, bottom=2pt, left=3pt, right=3pt,
    boxsep=0pt,
    before skip=9pt, after skip=9pt,
}
\tcolorboxenvironment{theorem}{
    enhanced jigsaw,
    colback=black!5,
    colframe=black!5,
    boxrule=0pt,

    before skip=9pt, after skip=9pt,
}
\tcolorboxenvironment{lemma}{

% --- Code & Algorithm Environments ---

\lstnewenvironment{codeexample}[1][]
{
    \refstepcounter{definition}
    \endgraf\vspace{2pt} 
    \noindent\textbf{Code Example \thedefinition}\endgraf\vspace{2pt} 
    \lstset{
        basicstyle=\ttfamily\scriptsize, 
        numbers=left,
        numberstyle=\tiny\color{black!50},
        stepnumber=1,
        numbersep=6pt,                  
        xleftmargin=18pt,               
        xrightmargin=2pt,               
        framexleftmargin=18pt,          
        framexrightmargin=2pt,          
        tabsize=4,
        showspaces=false,
        showstringspaces=false,
        breaklines=true,
        frame=single,
        rulecolor=\color{black!30},
        backgroundcolor=\color{black!2},
        aboveskip=2pt, belowskip=4pt,   
        breakatwhitespace=true,
        #1
    }
}
